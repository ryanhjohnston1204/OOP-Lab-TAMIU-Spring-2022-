#include <iostream>
using namespace std;

void grade(int score) {
	string x = "Student's Letter Grade: ";
	string list[] = { "A+", "A", "B+", "B", "C+",
		"C", "D+", "D", "F" };

	if (score >= 90 && score <= 100) {
		cout << x << list[0];
	}
	else if (score >= 85 && score <= 89) {
		cout << x << list[1];
	}
	else if (score >= 80 && score <= 84) {
		cout << x << list[2];
	}
	else if (score >= 75 && score <= 79) {
		cout << x << list[3];
	}
	else if (score >= 65 && score <= 69) {
		cout << x << list[4];
	}
	else if (score >= 60 && score <= 64) {
		cout << x << list[5];
	}
	else if (score >= 55 && score <= 59) {
		cout << x << list[6];
	}
	else if (score >= 50 && score <= 54) {
		cout << x << list[7];
	}
	else if (score < 50) {
		cout << x << list[8];
	}
	else {
		cout << "Invalid grade" << endl;
	}
}

int main() {
	int score;
	char ans;
	cout << "Enter a score: " << endl;
	cin >> score;
	grade(score);

	return 0;
}
