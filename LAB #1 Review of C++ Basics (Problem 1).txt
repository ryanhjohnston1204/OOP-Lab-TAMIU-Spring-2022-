#include <iostream>
#include <cmath>
using namespace std;
const double PI = 3.141592;// const variable pi 

void calc(double rad) {// calculates radius in 4 separate formulas
	double area = rad * PI, // AREA
		circ = (rad * 2) * PI, // CIRCUMFERENCE
		vol = (4 / 3 * PI) * pow(rad, 3), // VOLUME
		surf = 4 * PI * pow(rad, 2); // SURFACE AREA

	cout << "The area of a circle is: " << area << endl // Outputs AREA
		<< "The circumference of a circle is: " << circ << endl// Outputs CIRCUMFERENCE
		<< "The volume of the circle is: " << vol << endl// Outputs VOLUME
		<< "The surface area of the circle is: " << surf << endl;// Outputs SURFACE AREA
}

int main() {
	double rad;// variable radius declaration
	cout << "Enter a radius of a circle: " << endl;// output question for user input
	cin >> rad;// user input of radius
	calc(rad);//function call
	
	return 0;
}
